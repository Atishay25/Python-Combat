Emitted events\+: {\ttfamily add}, {\ttfamily add\+Dir}, {\ttfamily change}, {\ttfamily unlink}, {\ttfamily unlink\+Dir}, {\ttfamily all}, {\ttfamily error} \begin{DoxyVerb}new FSWatcher()

\end{DoxyVerb}
